class Solution {
public:
    //√Ê ‘∏ﬂ∆µ
    int FindGreatestSumOfSubArray(vector<int> array) {
        int res = INT_MIN, s = 0;
        //x 10  -11 3 4
        //s 10  -1  3 7
        //r 10   10 10 10
        for(auto x : array)
        {
            if(s < 0) s = 0;
            s += x;
            res = max(res, s);
        }
        
        return res;
    }
};

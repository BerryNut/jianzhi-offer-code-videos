class Solution {
public:
    static bool cmp(int a, int b)
    {
        string as = to_string(a), bs = to_string(b);
        //33 321
        //¡°33321¡± ¡°32133¡±
        return as + bs < bs + as;
    }
    
    string PrintMinNumber(vector<int> nums) {
        //33  321
        //33321 32133
        sort(nums.begin(), nums.end(),cmp);
        
        string res;
        for(auto x : nums) res += to_string(x);
        return res;
    }
};
